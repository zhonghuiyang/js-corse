    //实现一个简单的计算器  提示：定义三个变量   数字类型转换  功能：+、-、*、/
   var num1 = prompt('请输入数字?')
   var sy = prompt('请输入+、-、*、/')
   var num2 = prompt('请输入数字?')
   var res = null
   //做匹配
   switch(sy){
       case '+' :
           res = Number(num1) + Number(num2)
           break
       case '-' :
           res = num1 - num2
           break
       case '*' :
           res = num1 * num2
           break
       case '/' :
           res = num1 / num2
           break
   }
   document.write(res)
